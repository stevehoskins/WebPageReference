This is the class that does the calcs!

        
            
import java.util.*;
import java.lang.Math.*;

public class testing  {

    double length, width, areaOf, perimeterOf, radius;
    double circumference, areaOfCircle, volumeOfSphere; 

    public void calculateArea() {

        System.out.println("CalculateArea Method");
        System.out.println("");

        Scanner getInput = new Scanner(System.in);

        System.out.print("Enter width: ");
        width = Double.parseDouble(getInput.nextLine());

        System.out.print("Enter length: ");
        length = Double.parseDouble(getInput.nextLine());

        areaOf = length * width;

        System.out.println("Length is: " + length);
        System.out.println("Width is: " + width);
        System.out.println("Area: " + areaOf);
        System.out.println("");
    }

    public void calculatePerimeter() {

        System.out.println("CalculatePerimeter Method");
        System.out.println("");

        Scanner getInput = new Scanner(System.in);

        System.out.print("Enter width: ");
        width = Double.parseDouble(getInput.nextLine());

        System.out.print("Enter length: ");
        length = Double.parseDouble(getInput.nextLine());

        perimeterOf = 2 * (length + width);

        System.out.println("Length is: " + length);
        System.out.println("Width is: " + width);
        System.out.println("Perimeter: " + perimeterOf);
        System.out.println("");
    }

    public void calculateAreaOfCircle() {

        System.out.println("CalculateAreaOfCircle Method");
        System.out.println("");

        Scanner getInput = new Scanner(System.in);

        System.out.print("Enter radius: ");
        radius = Double.parseDouble(getInput.nextLine());

        areaOfCircle = Math.PI * Math.pow(radius, 2);

        System.out.println("Radius is: " + radius);
        System.out.println("Area of Circle: " + areaOfCircle);
        System.out.println("");
    }

    public void calculateCircumferenceOfCircle() {

        System.out.println("CalculateCircumferenceOfCircle Method");
        System.out.println("");

        Scanner getInput = new Scanner(System.in);

        System.out.print("Enter radius: ");
        radius = Double.parseDouble(getInput.nextLine());

        circumference = Math.PI * 2 * radius;

        System.out.println("Radius is: " + radius);
        System.out.println("Circumference of Circle: "
                             + circumference);
        System.out.println("");
    }

    public void volumeOfSpere() {

        System.out.println("volumeOfSphere Method");
        System.out.println("");

        Scanner getInput = new Scanner(System.in);

        System.out.print("Enter radius: ");
        radius = Double.parseDouble(getInput.nextLine());

        volumeOfSphere =
            (4.0 / 3.0 * Math.PI * radius * radius * radius);

        System.out.println("Radius is: " + radius);
        System.out.println("Volume of the Sphere: "
                                + volumeOfSphere);
        System.out.println("");
    }
}
